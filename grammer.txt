Annex A を参考に作成

program: <expression> ";" program
program: ε

<expression>:
  <assignment_expression>

<assignment_expression>:
  <equality_expression>
  <equality_expression> "=" <assignment_expression>

<equality_expression>:
  <additive_expression>
  <equality_expression> "==" <additive_expression>
  <equality_expression> "!=" <additive_expression>

<additive_expression>:
  <multiplicative_expression>
  <additive_expression> "+" <multiplicative_expression>
  <additive_expression> "-" <multiplicative_expression>

<multiplicative_expression>:
  <primary_expression>
  <multiplicative_expression> "*" <primary_expression>
  <multiplicative_expression> "/" <primary_expression>

<primary_expression>:
  <decimal-constant>
  <identifier>
  "(" <expression> ")"
